```env
# ==============================================================================
# TAURO PRODUCTION ENVIRONMENT VARIABLES

# Copy this file to .env.local (for development) or set in Vercel dashboard

# --- Mapbox (Property Maps) ---
NEXT_PUBLIC_MAPBOX_TOKEN=pk.eyJ1Ijo...                   # Mapbox public access token (https://account.mapbox.com/access-tokens/)

# --- GoHighLevel (CRM Integration) ---

GHL_LOCATION_ID=                                           # GHL location/sub-account ID (required if using API key)
GHL_WEBHOOK_SECRET=                                        # HMAC secret for verifying inbound GHL webhooks

# --- Cloudflare Turnstile (CAPTCHA) ---
NEXT_PUBLIC_TURNSTILE_SITE_KEY=0x4AAAAAAA...             # Turnstile site key (https://dash.cloudflare.com/turnstile)
TURNSTILE_SECRET_KEY=0x4AAAAAAA...                        # Turnstile secret key (server-side verification)

# --- Google Analytics ---
NEXT_PUBLIC_GA_ID=G-XXXXXXXXXX                            # GA4 measurement ID

# --- Sentry (Error Tracking) ---
NEXT_PUBLIC_SENTRY_DSN=https://xxx@sentry.io/xxx          # Sentry DSN for error reporting
SENTRY_AUTH_TOKEN=sntrys_eyJpYXQi...                      # Sentry auth token (for source map uploads during build)
SENTRY_ORG=your-org-slug                                  # Sentry organization slug
SENTRY_PROJECT=tauro                                      # Sentry project slug

# --- Upload Security ---
UPLOAD_API_KEY=                                            # API key for /api/upload endpoint (set to a random string)
```
