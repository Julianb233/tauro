export interface Neighborhood {
  id: string;
  name: string;           // Display name (e.g., "Center City")
  slug: string;           // URL slug (e.g., "center-city")
  tagline: string;        // One-line hook for cards (e.g., "The beating heart of Philadelphia")
  description: string;    // 2-3 paragraph rich description for the page (150-250 words)
  sellingPoints: string[];// 4-6 key selling points (bullet-style)
  lifestyle: {
    vibe: string;         // 1-2 sentences about the feel
    dining: string;       // Notable food/drink scene
    transit: string;      // Transportation highlights
    parks: string;        // Green spaces / outdoor
  };
  stats: {
    medianPrice: string;  // e.g., "$425,000"
    avgPricePerSqft: string; // e.g., "$285"
    avgDaysOnMarket: number;
    inventoryLevel: string; // e.g., "Low" | "Moderate" | "High"
  };
  image: string;          // Hero image URL (Unsplash, ?w=1200&q=80)
  cardImage: string;      // Card thumbnail URL (Unsplash, ?w=800&q=80)
  mapCenter: { lat: number; lng: number };
  propertyFilter: string; // Value to match against property.city for filtering
  featured: boolean;      // Whether shown on homepage (the 6 that are there now)
}
