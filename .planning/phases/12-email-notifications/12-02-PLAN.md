       ```typescript
       // 3. Send emails (best effort -- never block response on email failure)

       // 3a. Visitor confirmation email (all lead types except agent-application get lead confirmation)
       if (data.type !== "agent-application") {
         sendLeadConfirmation(data.email, {
           firstName: data.firstName,
           type: data.type,
           message: data.message,
         }).catch((err) => console.error("Lead confirmation email failed:", err));
       }

       // 3b. Agent-application gets application confirmation instead
       if (data.type === "agent-application") {
         sendApplicationConfirmation(data.email, {
           firstName: data.firstName,
           licenseNumber: data.licenseNumber,
         }).catch((err) => console.error("Application confirmation email failed:", err));
       }

       // 3c. Agent notification (or admin fallback)
       // Resolve agent email: if agentId was found, query their email; otherwise use ADMIN_EMAIL
       let notifyEmail: string | null = null;

       if (agentId) {
         const { data: agentRow } = await supabase
           .from("agents")
           .select("email")
           .eq("id", agentId)
           .single();
         notifyEmail = agentRow?.email ?? null;
       }

       // Fallback to admin for all lead types that need notification
       if (!notifyEmail) {
         notifyEmail = process.env.ADMIN_EMAIL || "admin@lylrealty.com";
       }

       // Send agent/admin notification for actionable lead types (not agent-application -- admin gets daily digest for those)
       if (data.type !== "agent-application") {
         sendAgentNotification(notifyEmail, {
           leadName: `${data.firstName} ${data.lastName}`,
           leadEmail: data.email,
           leadPhone: data.phone,
           leadType: data.type,
           message: data.message,
           propertyAddress: data.propertyAddress || data.homeAddress,
         }).catch((err) => console.error("Agent notification email failed:", err));
       }
       ```

    3. Update the comment numbering: the GHL webhook section (currently "2. Forward to GHL") becomes "4. Forward to GHL".

    IMPORTANT:
