    Create `src/lib/supabase/queries.ts` with reusable server-side query functions. These are used by both API routes and server components (in Plan 10-05).

    Each function creates a server client via `createClient()` from `./server` and queries the database.

    ```typescript
    import { createClient } from './server';

    // Properties
    export async function getProperties(options?: {
      neighborhood?: string;
      status?: string;
      propertyType?: string;
      minPrice?: number;
      maxPrice?: number;
      beds?: number;
      baths?: number;
      sortBy?: 'price' | 'created_at' | 'beds' | 'sqft';
      sortOrder?: 'asc' | 'desc';
      limit?: number;
      offset?: number;
      featured?: boolean;
    }) {
      const supabase = await createClient();
      let query = supabase.from('properties').select('*, agents!listing_agent_id(id, slug, full_name, photo, phone, email)', { count: 'exact' });

      if (options?.neighborhood) query = query.eq('neighborhood', options.neighborhood);
      if (options?.status) query = query.eq('status', options.status);
      if (options?.propertyType) query = query.eq('property_type', options.propertyType);
      if (options?.minPrice) query = query.gte('price', options.minPrice);
      if (options?.maxPrice) query = query.lte('price', options.maxPrice);
      if (options?.beds) query = query.gte('beds', options.beds);
      if (options?.baths) query = query.gte('baths', options.baths);
      if (options?.featured !== undefined) query = query.eq('featured', options.featured);

      const sortBy = options?.sortBy || 'created_at';
      const sortOrder = options?.sortOrder || 'desc';
      query = query.order(sortBy, { ascending: sortOrder === 'asc' });

      if (options?.limit) query = query.limit(options.limit);
      if (options?.offset) query = query.range(options.offset, options.offset + (options?.limit || 20) - 1);

      const { data, error, count } = await query;
      if (error) throw error;
      return { data: data || [], count: count || 0 };
    }

    export async function getPropertyBySlug(slug: string) {
      const supabase = await createClient();
      const { data, error } = await supabase
        .from('properties')
        .select('*, agents!listing_agent_id(id, slug, full_name, first_name, last_name, photo, phone, email, title)')
        .eq('slug', slug)
        .single();
      if (error) throw error;
      return data;
    }

    export async function getFeaturedProperties(limit = 6) {
      const supabase = await createClient();
      const { data, error } = await supabase
        .from('properties')
        .select('*, agents!listing_agent_id(id, slug, full_name, photo, phone, email)')
        .eq('featured', true)
        .limit(limit);
      if (error) throw error;
      return data || [];
    }

    // Agents
    export async function getAgents() { ... select all, order by created_at }
    export async function getAgentBySlug(slug: string) { ... select single with properties }

    // Neighborhoods
    export async function getNeighborhoods() { ... select all, order by name }
    export async function getNeighborhoodBySlug(slug: string) { ... select single }
    export async function getFeaturedNeighborhoods() { ... where featured = true }

    // Testimonials
    export async function getTestimonials() { ... select all, order by sort_order }

    // FAQ
    export async function getFaqs(category?: 'buyer' | 'seller' | 'general') { ... filter by category if provided, order by sort_order }
    ```

    export async function GET(request: NextRequest) {
      const { searchParams } = new URL(request.url);
      const status = searchParams.get('status');
      const type = searchParams.get('type');
      const limit = parseInt(searchParams.get('limit') || '50');
      const offset = parseInt(searchParams.get('offset') || '0');

      const supabase = await createClient();
      let query = supabase.from('leads').select('*', { count: 'exact' });
      if (status) query = query.eq('status', status);
      if (type) query = query.eq('type', type);
      query = query.order('created_at', { ascending: false }).range(offset, offset + limit - 1);

      const { data, error, count } = await query;
      if (error) return NextResponse.json({ error: error.message }, { status: 500 });
      return NextResponse.json({ data, count, limit, offset });
    }
